   Pozn. : V příkladu |30b| je demonstrován výběr "AND-BETWEEN"
SELECT cislo, nazev, cena
FROM Vyrobky
WHERE cena
BETWEEN 10 AND 20 OR cena BETWEEN 100 AND 140;
