"IN" a "OR" vrací stejné výsledky
SELECT cislo, nazev, sklad
FROM Vyrobky
WHERE sklad IN(2, 3);
SELECT cislo, nazev, sklad
FROM Vyrobky
WHERE sklad = 2 OR sklad = 3;
