Porovnat s příbuzným :
"IN" a "OR" vrací stejné výsledky
SELECT prijmeni, jmeno, pocet_deti
FROM zamestnanci
WHERE pocet_deti IN (1,3);
SELECT prijmeni, jmeno, pocet_deti
FROM zamestnanci
WHERE pocet_deti = 1 OR pocet_deti = 3;
